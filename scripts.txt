pipeline {
    agent any

    tools {
        maven 'MAVEN-HOME'
    }

    stages {

        stage('Clone Repo') {
            steps {
                // Delete old folder if exists
                bat 'if exist Smart-Assessment-Hub rmdir /S /Q Smart-Assessment-Hub'

                // Clone GitHub repo
                bat 'git clone https://github.com/krishnatejacoder/Smart-Assessment-Hub'
            }
        }

        stage('Clean') {
            steps {
                bat 'mvn -f Smart-Assessment-Hub/pom.xml clean'
            }
        }

        stage('Install') {
            steps {
                bat 'mvn -f Smart-Assessment-Hub/pom.xml install'
            }
        }

        stage('Test') {
            steps {
                bat 'mvn -f Smart-Assessment-Hub/pom.xml test'
            }
        }

        stage('Package') {
            steps {
                bat 'mvn -f Smart-Assessment-Hub/pom.xml package'
            }
        }
    }
}
